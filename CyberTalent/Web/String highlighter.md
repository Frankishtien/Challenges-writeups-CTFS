   	$('.sub').on('click',function(){
    		var color = $('select').val();
    		var txt = $('textarea').val();
    		var stringToParse = color + ':' + txt;
    		$.post('',{stp:stringToParse},function(data){
    			$('.preview').html(data);
    		});
    	});
